插件需继承 Plugin 基类并实现 execute 方法
# my_skill.py - 插件示例
from core.plugin_system import Plugin

class MySkill(Plugin):
    name = "my_skill"
    description = "我的自定义技能"
    version = "1.0.0"

    def execute(self, args):
        return f"结果: {args}"

    def get_schema(self):
        return {
            "name": self.name,
            "description": self.description,
            "parameters": {"type": "object", "properties": {}}
        }
